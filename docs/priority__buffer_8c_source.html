617 STATIC check_t overwriteElement(element_t element, priority_t lowestPri, priority_t newPriority)
618 {
619  check_t returnVal = INVALID_WRITE;
620  index_t tailIdx;
621  index_t lowestPriTailIdx = lowestPriorityTail();
622  index_t nextHighestHeadIdx;
623  index_t equalOrHigherPriHead;
624  priority_t nextHighestPri;
625  priority_t equalOrHigherPri;
626 
627  nextTailIndex(&tailIdx);
628 
629  // is buffer full of single priority?
630  if(solePriority(newPriority) == VALID_PRIORITY)
631  {
632  // buffer full of same priority
633  writeData(element, tailIdx);
634  setHead(tailIdx, newPriority);
635  incTail();
636  returnVal = VALID_WRITE;
637  }
638  else
639  {
640  // buffer full of multiple priorities
641  nextHighestPriority(&nextHighestPri, lowestPri);
642  nextHighestHeadIdx = headValue(nextHighestPri);
643  equalOrHigherPriority(&equalOrHigherPri, newPriority);
644  equalOrHigherPriHead = headValue(equalOrHigherPri);
645  writeData(element, lowestPriTailIdx);
646 
647  // is there a single element of the lowest priority and we are higher?
648  if((headValue(lowestPri) == lowestPriTailIdx) &&
649  (lowestPri != newPriority))
650  {
651  // overwrite single element and set inactive
652  setInactive(lowestPri);
653  setHead(lowestPriTailIdx, newPriority);
654  setActive(newPriority);
655  remap(equalOrHigherPriHead,nextHighestHeadIdx,headValue(newPriority));
656  writeTail(nextHighestHeadIdx);
657  }
658  else if((activeStatus(newPriority) == ACTIVE) &&
659  (lowestPri == newPriority))
660  {
661  // are there already entries of the same priority as the new entry
662  // where we are the lowest priority?
663  remap(nextHighestHeadIdx, lowestPriTailIdx, headValue(newPriority));
664  setHead(lowestPriTailIdx, newPriority);
665  writeTail(lowestPriTailIdx);
666  }
667  else
668  {
669  // we are adding the highest priority so insert after tail
670  nextHighestPriority(&nextHighestPri, lowestPri);
671  nextTailIndex(&tailIdx);
672  setHead(lowestPriTailIdx, newPriority);
673  remap(tailIndex(), nextHighestHeadIdx, lowestPriTailIdx);
674  }
675 
676  returnVal = VALID_WRITE;
677  }
678 
679  return returnVal;
680 }
